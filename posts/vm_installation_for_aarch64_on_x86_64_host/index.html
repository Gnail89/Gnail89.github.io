VM installation for aarch64 on x86_64 host#

案例一，centos 7运行aarch64虚拟机，完成ntp软件编译#

1. 基础环境#

  • 基础操作系统: CentOS 7 x86_64
  • 虚拟机操作系统: CentOS 7 aarch64
# 安装CentOS 7基础操作系统。并下载qemu源码包。
wget https://download.qemu.org/qemu-5.2.0.tar.xz      #qemu源码
wget http://releases.linaro.org/components/kernel/uefi-linaro/16.02/release/qemu64/QEMU_EFI.fd      #aarch64 UEFI引导程序

# 安装libvirt管理软件
yum install libvirt virt-manager
systemctl start libvirtd

2. 编译qemu#

yum install Ninja-build gcc gcc-c++ bzip2
tar xvJf qemu-5.2.0.tar.xz
cd qemu-5.2.0

./configure --target-list=aarch64-softmmu
make && make install

3. 安装aarch64虚拟机#

# 准备镜像：CentOS-7-aarch64-Minimal-2009.iso
# 创建卷
qemu-img create -f qcow2 centos7_aarch64.qcow2 10G

# 启动虚拟机
qemu-system-aarch64 -m 2048 -cpu cortex-a72 -smp 2 -M virt \
    -bios QEMU_EFI.fd -nographic \
    -drive if=none,file=centos7_aarch64.qcow2,id=hd0 \
    -device virtio-blk-device,drive=hd0 \
    -drive if=none,file=CentOS-7-aarch64-Minimal-2009.iso,id=cdrom,media=cdrom \
    -device virtio-scsi-device \
    -device scsi-cd,drive=cdrom \
    -net nic -net tap,ifname=tap0,script=no,downscript=no

4. 启动虚拟机#

qemu-system-aarch64 -m 2048 -cpu cortex-a72 -smp 2 -M virt \
    -bios QEMU_EFI.fd -nographic \
    -drive if=none,file=centos7_aarch64.qcow2,id=hd0 \
    -device virtio-blk-device,drive=hd0 \
    -device virtio-scsi-device \
    -net nic -net tap,ifname=tap0,script=no,downscript=no

5. 创建透明网桥#

# 将虚拟机网卡tap0加入br0，以实现网络互通
brctl addbr br0
brctl addif br0 eth0
brctl addif br0 tap0
ifup  tap0

添加网桥：brctl addbr br0
删除网桥：brctl delbr br0
显示网桥信息：brctl show
将eth0端口加入网桥br0：brctl addif br0 eth0
从网桥br0中删除eth0端口：brctl delif br0 eth0

6. 编译ntp软件#

wget http://archive.ntp.org/ntp4/ntp-4.2/ntp-4.2.8p15.tar.gz
tar zxf ntp-4.2.8p15.tar.gz
cd ntp-4.2.8p15

./configure --host=aarch64-linux-gnu --prefix=/opt/ntp-4.2.8p15 --enable-static
make && make install

案例二，OpenEuler 22.03运行aarch64虚拟机，zabbix agent编译#

1. 基础环境#

  • 基础操作系统: OpenEuler 22.03 x86_64
  • 虚拟机操作系统: alpine 3.23.3 aarch64
# 安装依赖
yum install libvirt qemu-system-aarch64

# 安装UEFI引导程序
wget https://dl-cdn.openeuler.openatom.cn/openEuler-22.03-LTS-SP3/everything/aarch64/Packages/edk2-aarch64-202011-13.oe2203sp3.noarch.rpm

# 解压包
rpm2cpio edk2-aarch64-202011-13.oe2203sp3.noarch.rpm | cpio -idmv

# 使用其中的QEMU_EFI-pflash.raw和vars-template-pflash.raw固件

2. 安装虚拟机#

# 创建卷
qemu-img create -f qcow2 /var/lib/libvirt/images/alpine-aarch64.qcow2 10G

# 通过virt-install启动虚拟机
virt-install \
  --name alpine-aarch64 \
  --arch aarch64 \
  --machine virt \
  --vcpus 2 \
  --cpu cortex-a72 \
  --memory 1024 \
  --os-variant alpinelinux3.12 \
  --disk path=/var/lib/libvirt/images/alpine-aarch64.qcow2,format=qcow2,size=10 \
  --cdrom /var/lib/libvirt/images/alpine-standard-3.23.3-aarch64.iso \
  --network network=default \
  --boot loader=/var/lib/libvirt/images/QEMU_EFI-pflash.raw,loader_ro=yes,loader_type=pflash,nvram_template=/var/lib/libvirt/images/vars-template-pflash.raw \
  --graphics vnc,listen=0.0.0.0 --noautoconsole

# 连接控制台执行安装
virsh console alpine-aarch64

# 停止虚拟机
virsh destroy alpine-aarch64

# 删除虚拟机
virsh undefine --nvram alpine-aarch64

3. 编译软件#

  • 基础配置
# 替换为国内软件源
echo "https://mirrors.ustc.edu.cn/alpine/v3.23/main
https://mirrors.ustc.edu.cn/alpine/v3.23/community" > /etc/apk/repositories

# 安装依赖软件
apk add --no-cache \
    make automake autoconf gcc libc-dev pkgconfig \
    tar g++ linux-headers \
    pcre2-dev pcre2-static
  • 编译openssl
tar -zxvf openssl-1.1.1w.tar.gz && \
    cd openssl-1.1.1w && \
    ./config --prefix=/opt/openssl no-shared no-zlib -static && \
    make -j$(nproc) && \
    make install
  • 编译zabbix agent
tar -zxvf zabbix-7.0.22.tar.gz
cd zabbix-7.0.22

./configure \
    --prefix=/opt/zabbix \
    --enable-agent \
    --enable-static \
    --with-libpcre2 \
    --enable-ipv6 \
    --with-openssl=/opt/openssl \
    CFLAGS="-static -std=gnu99" \
    LDFLAGS="-static"
make -j$(nproc) && make install
