自签名私有证书制作#

V1 版本#

  • ssl.key（私钥）
  • ssl.crt（证书）
  • dhparam.pem（DH 参数）文件
# 生成私钥文件 (ssl.key)
openssl genrsa -out ssl.key 2048

# 生成证书签名请求 (CSR)
openssl req -new -key ssl.key -out ssl.csr

# 生成自签名证书 (ssl.crt)
openssl x509 -req -days 365 -in ssl.csr -signkey ssl.key -out ssl.crt

#生成 DH 参数文件 (dhparam.pem)
openssl dhparam -out dhparam.pem 2048

# 验证私钥
openssl rsa -in ssl.key -check

# 查看证书详细信息
openssl x509 -in ssl.crt -text -noout

# 验证私钥和证书是否匹配, 如果两个哈希值相同，则私钥和证书匹配
openssl pkey -pubout -in ssl.key | openssl sha256
openssl x509 -pubkey -in ssl.crt -noout | openssl sha256

# 检查证书有效期
openssl x509 -in ssl.crt -noout -dates

# 验证DH参数
openssl dhparam -in dhparam.pem -check

V3 版本#

  • ssl.key（私钥）
  • ssl.crt（证书）
  • dhparam.pem（DH 参数）文件
# 创建 OpenSSL 配置文件 (ssl_v3.cnf)
cat > ssl_v3.cnf << "EOF"
[ req ]
default_bits        = 2048
distinguished_name  = req_distinguished_name
req_extensions      = v3_req
x509_extensions     = v3_ca
prompt              = no
default_md          = sha256

[ req_distinguished_name ]
C   = CN
ST  = State
L   = City
O   = Organization
OU  = Organizational Unit
CN  = example.com

[ v3_req ]
basicConstraints = CA:FALSE
keyUsage = nonRepudiation, digitalSignature, keyEncipherment
extendedKeyUsage = serverAuth, clientAuth
subjectAltName = @alt_names

[ v3_ca ]
basicConstraints = critical,CA:TRUE
subjectKeyIdentifier = hash
authorityKeyIdentifier = keyid:always,issuer:always
keyUsage = critical, digitalSignature, keyCertSign

[ alt_names ]
DNS.1   = localhost
DNS.2   = example.com
DNS.3   = www.example.com
IP.1    = 127.0.0.1
IP.2    = ::1
EOF

# 生成私钥文件 (ssl.key)
openssl genrsa -out ssl.key 2048

# 生成证书签名请求 (CSR)
openssl req -new -key ssl.key -out ssl.csr -config ssl_v3.cnf

# 生成自签名证书 (ssl.crt) (弃用扩展alt_names)
openssl x509 -req -in ssl.csr -signkey ssl.key -out ssl.crt -days 365 -sha256 -extfile ssl_v3.cnf

# 生成自签名证书 (ssl.crt) (引入扩展alt_names)
openssl x509 -req -in ssl.csr -signkey ssl.key -out ssl.crt -days 365 -sha256 -extfile ssl_v3.cnf -extensions v3_req

#生成 DH 参数文件 (dhparam.pem)
openssl dhparam -out dhparam.pem 2048

# 查看证书详细信息
openssl x509 -in ssl.crt -text -noout

测试实例#

# 1. 创建 OpenSSL 配置文件 (ssl_v3.cnf)
cat > ssl_v3.cnf << "EOF"
[ req ]
default_bits        = 2048
distinguished_name  = req_distinguished_name
req_extensions      = v3_req
x509_extensions     = v3_ca
prompt              = no
default_md          = sha256

[ req_distinguished_name ]
C   = CN
ST  = State
L   = City
O   = Organization
OU  = Organizational Unit
CN  = example.com

[ v3_req ]
basicConstraints = CA:FALSE
keyUsage = nonRepudiation, digitalSignature, keyEncipherment
extendedKeyUsage = serverAuth, clientAuth
subjectAltName = @alt_names

[ v3_ca ]
basicConstraints = critical,CA:TRUE
subjectKeyIdentifier = hash
authorityKeyIdentifier = keyid:always,issuer:always
keyUsage = critical, digitalSignature, keyCertSign

[ alt_names ]
DNS.1   = localhost
DNS.2   = example.com
DNS.3   = www.example.com
IP.1    = 127.0.0.1
IP.2    = ::1
EOF

# 2. 生成CA私钥文件
openssl genrsa -out ca.key 2048

# 3. 生成CA证书签名请求
openssl req -new -key ca.key -out ca.csr

# 4. 生成CA自签名证书
openssl x509 -req -in ca.csr -signkey ca.key -out ca.crt -days 3650 -sha256 -extfile ssl_v3.cnf -extensions v3_ca

# 5. 生成服务器私钥文件
openssl genrsa -out ssl.key 2048

# 6. 生成服务器证书签名请求
openssl req -new -key ssl.key -out ssl.csr -config ssl_v3.cnf -extensions v3_req

# 7. 签名服务器证书
openssl x509 -req -in ssl.csr -CA ca.crt -CAkey ca.key -CAcreateserial -out ssl.crt -days 365 -sha256 -extfile ssl_v3.cnf -extensions v3_req

# 8. 生成 DH 参数文件 (dhparam.pem)
openssl dhparam -out dhparam.pem 2048

# 9. 查看证书详细信息
openssl x509 -in ssl.crt -text -noout
